# JetMeAway — robots directives
#
# Public travel comparison site. We want BOTH search crawlers
# and LLM retrieval bots to read everything. Explicit allows are
# spelled out because some AI crawlers default to blocked unless
# the site opts in.

User-agent: *
Allow: /
Disallow: /api/
Disallow: /admin/
Disallow: /checkout/
Disallow: /hotels/checkout/
Disallow: /thank-you
Disallow: /success
Disallow: /redirect
